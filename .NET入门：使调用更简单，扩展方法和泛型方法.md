public class StringAnalyzer
{
    public static int CountA(string str)
    {
        if (string.IsNullOrEmpty(str))
            return 0;

        int count = 0;
        foreach (char c in str)
        {
            if (char.ToUpper(c) == 'A')
                count++;
        }
        return count;
    }
}
